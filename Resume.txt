    <script>

      let API = "https://api.github.com/users";

      async function getData() {
        let data = await fetch(API);
        let result = await data.json();

        let rows = "";

        result.forEach((user) => {
          rows += `
              <tr>
                <td>${user.id}</td>
                <td>${user.login}</td>
                <td>
                  <img src="${user.avatar_url}" width="50" height="50">
                </td>
              </tr>`;
        });

        document.getElementById("tableBody").innerHTML = rows;
      }

      getData();
    </script>
